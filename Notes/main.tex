\section*{Description of Heat Transfer Process}
Sunlight heats the outer surface of the black powder-coated Hydro Flask. Heat is transferred via multiple modes:
\begin{itemize}
    \item \textbf{Radiation:} Sunlight (visible and infrared) is absorbed by the black coated surface.
    \item \textbf{Conduction:} Heat conducts rapidly through the thin stainless steel wall into the water.
    \item \textbf{Convection:} Ambient air convects heat away from the bottle surface.  
    \textit{(No internal convection inside the water is considered; water is assumed well-mixed.)}
\end{itemize}

\section*{Engineering Problem Statement}
Determine the \textbf{steady-state water temperature} \(T_w\) in a 500 mL black Hydro Flask exposed to sunlight.

\textbf{Given:}
\begin{itemize}
    \item Solar irradiance: \(q_\text{solar} = 800~\text{W/m²}\)
    \item Ambient temperature: \(T_\infty = 300~\text{K}\)
    \item Bottle surface absorptivity: \(\alpha = 0.85\)
    \item Bottle emissivity: \(\epsilon = 0.9\)
    \item Convective heat transfer coefficient (air): \(h = 10~\text{W/m²·K}\)
    \item Wall thickness: \(L = 0.001~\text{m}\), stainless steel thermal conductivity: \(k = 16~\text{W/m·K}\)
    \item Bottle surface area: \(A = 0.03~\text{m²}\)
\end{itemize}

\textbf{Find:} Water temperature \(T_w\) at steady state if the bottle surface temperature \(T_s\) is assumed to be \(330~\text{K}\).

\section*{Assumptions}
\begin{enumerate}
    \item Steady-state conditions.  
    \item Bottle wall temperature is uniform (lumped capacitance approximation).  
    \item Properties (k, h, $\alpha$, $\epsilon$) are constant.  
    \item Neglect heat losses through bottle cap and edges.  
    \item Thin stainless steel wall allows conduction resistance to be small compared to convection.
    \item Water is well-mixed (isothermal), so internal temperature gradients are neglected.
    \item Water temperature equals the inner wall temperature (perfect conduction).
    \item Surface is diffuse and opaque.  
\end{enumerate}

\section*{Problem Setup / Equations}
Energy balance on the water:
\begin{equation}
q_\text{solar} \cdot \alpha A = q_\text{conduction} + q_\text{convection} + q_\text{radiation}
\end{equation}

\subsection*{Heat Transfer Relations}
\begin{align}
\text{Conduction through bottle wall: } & q_\text{cond} = \frac{k A (T_s - T_w)}{L} \\
\text{Convection to ambient air: } & q_\text{conv} = h A (T_s - T_\infty) \\
\text{Radiation from surface to surroundings: } & q_\text{rad} = \epsilon \sigma A (T_s^4 - T_\infty^4)
\end{align}

Assuming thin stainless steel wall (negligible conduction resistance) and well-mixed water:
\begin{equation}
q_\text{solar} \cdot \alpha - q_\text{rad} = q_\text{conv} \implies
T_w \approx T_s - \frac{\alpha q_\text{solar} - \epsilon \sigma (T_s^4 - T_\infty^4)}{h}
\end{equation}

\section*{Solution of the Equation}
Assume bottle surface temperature \(T_s = 330~\text{K}\):
\begin{align*}
q_\text{rad} &= \epsilon \sigma (T_s^4 - T_\infty^4) \\
&= 0.9 \cdot 5.67\times10^{-8} \cdot (330^4 - 300^4) \\
330^4 &= 1.19\times10^{10}, \quad 300^4 = 8.1\times10^9 \\
\Delta &= 3.8\times10^9 \\
q_\text{rad} &\approx 0.9 \cdot 5.67\times10^{-8} \cdot 3.8\times10^9 \approx 194~\text{W/m²} \\
q_\text{net} &= \alpha q_\text{solar} - q_\text{rad} = 0.85\cdot 800 - 194 \approx 494~\text{W/m²} \\
T_w &= T_s - \frac{q_\text{net}}{h} = 330 - \frac{494}{10} \approx 281~\text{K} \approx 8^\circ\text{C}
\end{align*}

\textit{Note:} Water is assumed isothermal (well-mixed), so conduction through the wall is fast and internal temperature gradients are neglected.

\section*{Reasonableness and Insights}
\textbf{Reasonableness:}
\begin{itemize}
    \item Temperature below surface temp — logical.  
    \item Magnitude reasonable for a small Hydro Flask in direct sun.  
\end{itemize}

\textbf{Insights:}
\begin{itemize}
    \item Radiation drives heating; convection removes heat to ambient air.  
    \item Thin stainless steel wall minimizes conduction resistance.  
    \item Dark powder coating maximizes solar absorption and heating efficiency.  
\end{itemize}

\begin{abstract}
Smartphones operating under sustained computational loads, such as navigation
applications in high ambient temperatures, are susceptible to overheating.
This report investigates the feasibility of using a pin-fin heat sink attached
to the rear surface of a smartphone to reduce operating temperature. Analytical
fin theory is applied to evaluate the effects of fin material, fin geometry, and
array configuration on heat dissipation performance.
\end{abstract}

% -----------------------------
\section{Introduction}

Modern smartphones generate significant heat during extended use. Excessive temperature
rise can degrade performance, reduce battery life, and cause user discomfort.
Passive cooling solutions such as pin-fin heat sinks offer a low-power method
to enhance convective heat transfer.

This study analyzes a pin-fin heat sink design attached to the back of a smartphone
and evaluates how fin length, material, and array configuration influence the phone
surface temperature.

% -----------------------------
\section{Physical Model and Assumptions}

\begin{itemize}
    \item Steady-state heat transfer
    \item Uniform base temperature across the phone surface
    \item Negligible thermal resistance between phone and fin base
    \item Adiabatic fin tips
    \item Constant convection coefficient
    \item Radiation effects neglected
\end{itemize}

Phone base area: $A_b$, heat dissipation rate: $Q$, ambient temperature: $T_\infty$.

% -----------------------------
\section{Analytical Formulation}

Corrected fin length for adiabatic tip:

\begin{equation}
L_c = L + \frac{D}{4}
\end{equation}

Fin parameter:

\begin{equation}
m = \sqrt{\frac{4h}{kD}}
\end{equation}

Fin efficiency:

\begin{equation}
\eta_f = \frac{\tanh(m L_c)}{m L_c}
\end{equation}

Base temperature:

\begin{equation}
T_b = T_\infty + \frac{Q}{h A_{b,f} + N \eta_f h A_f}
\end{equation}

\section{Conclusions}

Pin-fin heat sinks can significantly reduce smartphone surface temperature under
steady-state conditions. High-conductivity materials such as Aluminum 2024-T6
perform best. The optimal design is 20 fins, 3\,mm diameter, balancing heat dissipation
and manufacturing constraints.

\section{Appendix: Python Code}

\begin{lstlisting}
import numpy as np
import matplotlib.pyplot as plt

# Enable interactive plotting
plt.ion()

# Given constants
Q_total = 10.0          # W
T_inf = 20.0            # C
h = 50.0                # W/m^2*K
A_base = 0.06 * 0.124   # m^2

fin_lengths = np.linspace(0.001, 0.1, 100)

# Material properties
materials = {
    "Aluminum 2024-T6": 177,
    "AISI 302 Stainless Steel": 15.1,
    "White Pine Wood": 0.11,
    "Clay": 1.3
}

# Fin arrays
fin_arrays = {
    "10 fins, 5mm dia": (10, 0.005),
    "20 fins, 3mm dia": (20, 0.003),
    "30 fins, 1mm dia": (30, 0.001)
}

def phone_temp(Q, k, L, h, N, D, A_base, T_inf):
    Lc = L + D/4
    m = np.sqrt(4*h/(k*D))
    eta = np.tanh(m*Lc)/(m*Lc)
    A_fin = np.pi * D * Lc
    A_base_free = max(A_base - N*np.pi*(D/2)**2, 0.0)
    T_base = T_inf + Q/(h*A_base_free + N*eta*h*A_fin)
    return T_base, eta

# 1) Phone temp vs fin length (materials)
array_choice = "20 fins, 3mm dia"
N_fins, D = fin_arrays[array_choice]

plt.figure()
for material, k in materials.items():
    temps = []
    for L in fin_lengths:
        T_base, _ = phone_temp(Q_total, k, L, h, N_fins, D, A_base, T_inf)
        temps.append(T_base)
    plt.plot(fin_lengths*1000, temps, label=material)
plt.xlabel("Fin Length (mm)")
plt.ylabel("Phone Temperature ($^\circ$ C)")
plt.legend()
plt.grid(True)
plt.savefig("figures/materials_vs_length.png", dpi=300)

# 2) Phone temp vs fin arrays
material_choice = "Aluminum 2024-T6"
k = materials[material_choice]

plt.figure()
for array_name, (N_fins, D) in fin_arrays.items():
    temps = []
    for L in fin_lengths:
        T_base, _ = phone_temp(Q_total, k, L, h, N_fins, D, A_base, T_inf)
        temps.append(T_base)
    plt.plot(fin_lengths*1000, temps, label=array_name)
plt.xlabel("Fin Length (mm)")
plt.ylabel("Phone Temperature ($^\circ$ C)")
plt.legend()
plt.grid(True)
plt.savefig("figures/length_vs_phone_temp_arrays.png", dpi=300)

# 3) Optimal design
N_opt, D_opt = fin_arrays["20 fins, 3mm dia"]
k_opt = materials["Aluminum 2024-T6"]

temps_opt = []
eff_opt = []

for L in fin_lengths:
    T_base, eta = phone_temp(Q_total, k_opt, L, h, N_opt, D_opt, A_base, T_inf)
    temps_opt.append(T_base)
    eff_opt.append(eta)

plt.figure()
plt.plot(fin_lengths*1000, temps_opt)
plt.xlabel("Fin Length (mm)")
plt.ylabel("Phone Temperature ($^\circ$ C)")
plt.grid(True)
plt.savefig("figures/temps_vs_length_optimal.png", dpi=300)

# 4) Fin efficiency
plt.figure()
plt.plot(fin_lengths*1000, eff_opt)
plt.xlabel("Fin Length (mm)")
plt.ylabel("Fin Efficiency")
plt.grid(True)
plt.savefig("figures/efficiency_vs_length.png", dpi=300)

input("Press Enter to close all plots...")
plt.close("all")
